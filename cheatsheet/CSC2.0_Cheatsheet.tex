\section{Quick Hacks}
\cm{Ctrl+C }{Kills Current Command}
\cm{Ctrl+D }{Exit or Finished}
\cm{$\uparrow$ }{Scroll though history }
\cm{sudo !! }{Execute last command as root }
\cm{Ctrl+Shift+C }{Copy}
\cm{Ctrl+Shift+V }{Paste}
\cm{. }{Current directory}
\cm{.. }{Up one directory}
\cm{./<command> }{Execute command in current dir}

\section{Monday}
\cm{ls }{list files in directory}
\cm{cd <directory>}{change directory}
\cm{whoami }{current username}
\cm{alias <command>='<newcommand'> }{alias command}
\cm{pwd }{print working directory}
\cm{cat <file> }{read file}
\cm{cp  <file> <copy\_of\_file> }{copy file}
\cm{mv  <file> <newfile> }{move file}
\cm{rm  <file> }{remove file - forever!}
\cm{mkdir  <dir\_name> }{make directory}
\cm{grep {\it pattern}  <file> }{search file for pattern}
\cm{man <command>}{the manual for $<$command$>$}
\cm{sudo <command>}{EUID of 0 - root for $<$command$>$}
\cm{ps }{list current processes, -e or aux }
\cm{kill <pid> }{kill process $<$pid$>$}

\section{Tuesday}
\cm{apt-get install {\it app}}{install {\it app}}
\cm{adduser {\it username}}{add user {\it username}}
\cm{deluser {\it username}}{delete user {\it username}}
\cm{passwd}{change password}
\cm{ifconfig }{configure a network interface}
\cm{ssh -X -C <username>$@$<ip\_addr> }{SSH to $<$ip\_addr$>$ as $<$username$>$}
\cm{ssh-keygen}{generate an SSH key}
\cm{ssh-copy-id <username>@<ip\_addr> }{copy SSH key}
\cm{netstat -lntu}{list all open ports}
\cm{nmap localhost}{list all open ports}
\cm{w}{who is logged on and what they are doing}
\cm{who}{who is logged on}
\cm{whoami}{print EUID}
\cm{lastlog}{report recent user login}
\cm{reboot}{restart computer}
\cm{pkill}{kill processes based on name}

\section{Wednesday}
\cm{UFW}{firewall}
\cm{/etc/issue.net}{SSH Banner}
\cm{/var/log/auth.log}{SSH Log}
\cm{/var/www/html/index.html}{webpage}
\cm{service <service> start/stop/resart }{services}


\section{Thursday}
\cm{\#!/bin/bash}{start a script}
\cm{cron}{schedule commands}
\cm{chattr}{change file attributes}
\cm{lsattr}{list file attributes}
